## Installation

```bash
npx shadcn@latest add "https://agw-reusables.vercel.app/r/abstract-profile.json"
```

## Usage

The component automatically uses the connected wallet address:

```tsx
import { AbstractProfile } from "@/components/abstract-profile"

export default function AbstractProfileDemo() {
  return (
    <AbstractProfile />
  )
}
```

### Custom Address

Display a profile for a specific wallet address:

```tsx
import { AbstractProfile } from "@/components/abstract-profile"

export default function CustomAddressProfile() {
  return (
    <AbstractProfile address="0x1234567890123456789012345678901234567890" />
  )
}
```

### Size Variants

The component supports three size variants:

```tsx
import { AbstractProfile } from "@/components/abstract-profile"

export default function ProfileSizes() {
  return (
    <div className="flex items-center gap-4">
      <AbstractProfile size="sm" />
      <AbstractProfile size="md" />
      <AbstractProfile size="lg" />
    </div>
  )
}
```

### Custom Styling

Override the tier-based border color with a custom color:

```tsx
import { AbstractProfile } from "@/components/abstract-profile"

export default function CustomStyledProfile() {
  return (
    <AbstractProfile 
      shineColor="#8B5CF6" 
      fallback="JS" 
    />
  )
}
```

### Without Tooltip

Disable the tooltip that shows the display name:

```tsx
import { AbstractProfile } from "@/components/abstract-profile"

export default function ProfileWithoutTooltip() {
  return (
    <AbstractProfile showTooltip={false} />
  )
}
```

### Using Hooks Directly

Access profile data directly with the provided hooks:

<CodeCollapsibleWrapper>
```tsx
"use client"

import { useAbstractProfile, useAbstractProfileByAddress } from "@/hooks/use-abstract-profile"
import { useAccount } from "wagmi"
import { getTierColor, getTierName } from "@/lib/tier-colors"

export default function ProfileDataDemo() {
  const { address } = useAccount()
  const { data: profile, isLoading } = useAbstractProfile()
  const { data: otherProfile } = useAbstractProfileByAddress("0x1234567890123456789012345678901234567890")

  return (
    <div className="space-y-4">
      <div>
        <h3>Your Profile</h3>
        {isLoading ? (
          <p>Loading...</p>
        ) : profile ? (
          <div className="space-y-2">
            <p>Name: {profile.user?.name || "Anonymous"}</p>
            <p>Tier: {getTierName(profile.user?.tier || 1)}</p>
            <div 
              className="w-4 h-4 rounded-full border-2" 
              style={{ borderColor: getTierColor(profile.user?.tier || 1) }}
            />
          </div>
        ) : (
          <p>No profile found</p>
        )}
      </div>
    </div>
  )
}
```
</CodeCollapsibleWrapper>

## What's included

This command installs the following files:

- `/components`
  - [`abstract-profile.tsx`](#abstract-profiletsx) - The main profile component with tier-based styling

- `/hooks`
  - [`use-abstract-profile.ts`](#use-abstract-profilets) - React Query hooks for fetching profile data

- `/lib`
  - [`get-user-profile.ts`](#get-user-profilets) - API utility for fetching user profiles from Abstract Portal
  - [`tier-colors.ts`](#tier-colorsts) - Tier color mapping and utilities
  - [`address-utils.ts`](#address-utilsts) - Address formatting utilities

### `abstract-profile.tsx`

A profile component that displays user profile pictures from Abstract Portal with automatic tier-based border styling, loading states, and fallback support.

<ComponentSource src="registry/new-york/blocks/abstract-profile/abstract-profile.tsx" />

#### Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `address` | `string` | `undefined` | Optional wallet address to display profile for (defaults to connected wallet) |
| `fallback` | `string` | First 2 chars of address | Fallback text to display if image fails to load |
| `shineColor` | `string` | Tier-based color | Custom border color (overrides tier color) |
| `size` | `"sm" \| "md" \| "lg"` | `"md"` | Avatar size variant |
| `showTooltip` | `boolean` | `true` | Whether to show tooltip with display name on hover |
